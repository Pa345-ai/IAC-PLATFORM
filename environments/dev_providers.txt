# environments/dev/providers.tf

provider "aws" {
  region = var.aws_region
  
  default_tags {
    tags = {
      Environment = "dev"
      ManagedBy   = "terraform"
      Repository  = "sovereign-cloud-archive"
      CostCenter  = "development"
    }
  }
}

variable "aws_region" {
  description = "AWS region"
  type        = string
  default     = "us-east-1"
}

variable "tfstate_bucket" {
  description = "S3 bucket for Terraform state"
  type        = string
}

variable "vpc_cidr" {
  description = "CIDR block for dev VPC"
  type        = string
  default     = "10.2.0.0/16"
}
