# environments/global/iam/variables.tf

variable "aws_region" {
  description = "AWS region"
  type        = string
  default     = "us-east-1"
}

variable "allowed_regions" {
  description = "List of allowed AWS regions for resource creation"
  type        = list(string)
  default = [
    "us-east-1",
    "us-east-2",
    "us-west-1",
    "us-west-2"
  ]
}

variable "admin_external_id" {
  description = "External ID for admin role assumption (store in Secrets Manager)"
  type        = string
  sensitive   = true
}

variable "auditor_external_id" {
  description = "External ID for auditor role assumption"
  type        = string
  sensitive   = true
}

variable "auditor_account_arns" {
  description = "List of AWS account ARNs allowed to assume auditor role"
  type        = list(string)
  default     = []
}

variable "default_tags" {
  description = "Default tags for all resources"
  type        = map(string)
  default = {
    ManagedBy  = "terraform"
    Layer      = "global"
    Component  = "iam"
    Compliance = "pci-dss-nist"
  }
}
